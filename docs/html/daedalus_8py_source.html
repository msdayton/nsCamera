269 def setInterlacing(self, ifactor=None, side=None):
270 """
271 Sets interlacing factor. NOTE: if called directly when HFW or ZDT mode is active,
272 this will disengage those modes automatically. If hemispheres have different
273 factors when the image is acquired, the resulting frames are separated into
274 half-width images
275
276 Args:
277 ifactor: number of interlaced lines (generates ifactor + 1 images per frame)
278 defaults to 0 (no interlacing)
279 side: identify particular hemisphere (A or B) to control. If left blank,
280 control both hemispheres
281
282 Returns:
283 integer: active interlacing factor (unchanged if error)
284 """
285 logging.debug(self.logdebug + "setInterlacing; ifactor = " + str(ifactor))
286 if ifactor is None:
287 ifactor = 0
288 if (
289 not isinstance(ifactor, int)
290 or ifactor < 0
291 or ifactor > (self.maxheight - 1)
292 ):
293 err = (
294 self.logerr + "invalid interlacing factor submitted. "
295 "Interlacing remains unchanged. "
296 )
297 logging.error(err)
298 return self.interlacinginterlacing
299 if self.HFWHFW:
300 logging.warning(
301 self.logwarn + "HFW mode will be disengaged because of new "
302 "interlacing setting "
303 )
304 self.setHighFullWellsetHighFullWell(False)
305 if self.ZDTZDT:
306 logging.warning(
307 self.logwarn + "ZDT mode will be disengaged because of new "
308 "interlacing setting "
309 )
310 self.setZeroDeadTimesetZeroDeadTime(False)
311 if ifactor == 0:
312 bitscheme = self.maxheight * [0]
313 # deactivating one side shouldn't turn off enable for both sides
314 # TODO: is it a problem if sides are set separately, so interlacing is zero
315 # but still enabled?
316 if side is None:
317 self.caca.setSubregister("INTERLACING_EN", "0")
318 else:
319 pattern = [0] + ifactor * [1]
320 reps = 1 + self.maxheight // (ifactor + 1)
321 bitscheme = (reps * pattern)[0 : self.maxheight]
322 self.caca.setSubregister("INTERLACING_EN", "1")
323 err = ""
324 for regnum in range(32):
325 regbits = bitscheme[32 * regnum : 32 * (regnum + 1)]
326 logging.debug(self.logdebug + "regbits = " + str(regbits))
327 # generated pattern is reverse order from placement in register (element 0
328 # of the list is the LSB of the register)
329 bitsrev = regbits[::-1]
330 s = [str(i) for i in bitsrev]
331 b = "".join(s) # assemble as binary number for processing
332 hexval = "%x" % int(b, 2)
333 val = hexval.zfill(8)
334 err0 = ""
335 err1 = ""
336 if side is None or side.lower() == "a":
337 lname = "RSL_CONFIG_DATA_A" + str(regnum)
338 err1, _ = self.caca.setRegister(lname, val)
339 self.interlacinginterlacing[1] = ifactor
340 if side is None or side.lower() == "b":
341 rname = "RSL_CONFIG_DATA_B" + str(regnum)
342 err0, _ = self.caca.setRegister(rname, val)
343 self.interlacinginterlacing[0] = ifactor
344 err = err + err0 + err1
345 if err:
346 logging.error(self.logerr + "interlacing may not be set correctly: " + err)
347 logging.info(self.loginfo + "Interlacing set to " + str(self.interlacinginterlacing))
348 if self.interlacinginterlacing[0] == self.interlacinginterlacing[1]:
349 self.columnscolumns = 1
350 else:
351 self.columnscolumns = 2
352 return self.interlacinginterlacing

474 def setTriggerDelay(self, delay=0):
475 """
476 Use trigger delay timer. Actual delay is rounded down to multiple of .15 ns, up
477 to a maximum delay of 6 ns
478
479 Args:
480 delay: trigger delay in ns
481
482 Returns:
483 String of errors, if any
484 """
485 logging.debug(self.logdebug + "setTriggerDelay; delay = " + str(delay))
486 if (
487 not (isinstance(delay, int) or isinstance(delay, float))
488 or delay < 0
489 or delay > 6
490 ):
491 err = (
492 self.logerr + "invalid trigger delay submitted. Delay remains "
493 "unchanged. "
494 )
495 logging.error(err)
496 return err
497 delayblocks = int(delay / 0.15)
498 if delayblocks < 0:
499 delayblocks = 0
500 if delayblocks > 40:
501 delayblocks = 40
502 delayseq = (40 - delayblocks) * [0] + delayblocks * [1]
503 seqstr = "".join(str(x) for x in delayseq)
504 seqhex = "%x" % int(seqstr, 2)
505 logging.debug(self.logdebug + "seqhex = " + str(seqhex))
506 highpart = seqhex[-10:-8].zfill(8)
507 lowpart = seqhex[-8:].zfill(8)
508 self.caca.setSubregister("HST_DEL_EN", "1")
509 err0, _ = self.caca.setRegister("HST_TRIGGER_DELAY_DATA_LO", lowpart)
510 err1, _ = self.caca.setRegister("HST_TRIGGER_DELAY_DATA_HI", highpart)
511 err2, _ = self.caca.setSubregister("HST_MODE", "1")
512 delayed = delayblocks * 0.15
513 logging.info(self.loginfo + "Actual trigger delay = " + str(delayed) + " ns")
514 return err0 + err1 + err2

608 def setManualTiming(self, timing=None):
609 """
610 Manual shutter timing, five intervals given in nanoseconds, e.g.,
611 [100,50,100,50,100] for frame 0 open for 100 ns, an interframe pause of 50 ns,
612 frame 1 open for 100 ns, etc. Timing is set for both hemispheres.
613
614 The actual timing is rounded down to the nearest multiple of 25 ns. (Each
615 count = 25 ns. e.g., a request for 140 ns rounds down to a count of '5',
616 which corresponds to 125 ns))
617 - Minimum timing is 75 ns
618 - Maximum is 25 * 2^30 ns (approximately 27 seconds)
619
620 Args:
621 timing: 5-element list in nanoseconds
622
623 Returns:
624 tuple (error string, response string from final message)
625 """
626 if timing is None:
627 logging.info(
628 self.loginfo
629 + "No manual timing setting provided, defaulting to (100, 150, 100, "
630 " 150, 100, 150, 100) for both hemispheres"
631 )
632 timing = [(100, 150, 100, 150, 100)]
633
634 logging.info(self.loginfo + "Manual shutter sequence: " + str(timing))
635 flattened = flattenlist(timing)
636 if (
637 len(flattened) != 5
638 or not all(isinstance(x, (int, float)) for x in flattened)
639 or not all(x >= 25 for x in flattened)
640 ):
641 err = self.logerr + "Invalid manual shutter timing list: " + str(timing)
642 logging.error(err + "; timing settings unchanged")
643 return err, "00000000"
644
645 timecounts = [int(a // 25) for a in flattened]
646 self.caca.sensmanual = timing
647 self.caca.senstiming = {} # clear HST settings from ca object
648
649 control_messages = [
650 ("MANSHUT_MODE", "1"),
651 ("EXT_PHI_CLK_SH0_ON", "{0:#0{1}x}".format(timecounts[0], 10)[2:10]),
652 ("EXT_PHI_CLK_SH0_OFF", "{0:#0{1}x}".format(timecounts[1], 10)[2:10]),
653 ("EXT_PHI_CLK_SH1_ON", "{0:#0{1}x}".format(timecounts[2], 10)[2:10]),
654 ("EXT_PHI_CLK_SH1_OFF", "{0:#0{1}x}".format(timecounts[3], 10)[2:10]),
655 ("EXT_PHI_CLK_SH2_ON", "{0:#0{1}x}".format(timecounts[4], 10)[2:10]),
656 ]
657 return self.caca.submitMessages(control_messages, " setManualShutters: ")

677 def getSensTemp(self, scale=None, offset=None, slope=None, dec=1):
678 """
679 Read temperature sensor located on the Daedalus sensor
680 Args:
681 scale: temperature scale to report (defaults to C, options are F and K)
682 offset: offset of linear fit of sensor response (defaults to self.toffset)
683 slope: slope of linear fit of sensor response (defaults to self.tslope)
684 dec: round to 'dec' digits after the decimal point
685
686 Returns:
687 temperature as float on given scale, rounded to .1 degree
688 """
689 err, rval = self.caca.getMonV("MON_TSENSE_OUT", errflag=True)
690 if err:
691 logging.error(
692 self.logerr + "unable to retrieve temperature information ("
693 'getTemp), returning "0" '
694 )
695 return 0.0
696 if offset is None:
697 offset = self.toffset
698 if slope is None:
699 slope = self.tslope
700
701 ctemp = offset + slope * rval
702 if scale == "K":
703 temp = round(ctemp + 273.15, dec)
704 elif scale == "F":
705 temp = round(1.8 * ctemp + 32, dec)
706 else:
707 temp = round(ctemp, dec)
708 return temp

710 def parseReadoff(self, frames, columns):
711 """
712 Parses frames from board into images
713 Args:
714 frames: list of data arrays (frames) returned from board
715 columns: 1 (full width image) or 2 (hemispheres generate distinct images)
716 Returns:
717 list of data arrays (frames) reordered and deinterlaced
718 """
719 logging.debug(self.logdebug + "parseReadoff")
720 w = self.width
721 if hasattr(self, "ca"): # TODO: this may no longer be necessary
722 padIt = self.caca.padToFull
723 else:
724 padIt = self.padToFull
725 if padIt:
726 rows = self.maxheight
727 else:
728 rows = self.lastrow - self.firstrow + 1
729 parsed = []
730 for frame in frames:
731 current = np.zeros((rows, w), dtype=np.uint16)
732 mapped = np.zeros((rows, w), dtype=np.uint16)
733 frame = frame.reshape(rows, w)
734
735 for entry in range(int(w / 2)):
736 col = 32 * (entry % 8) + entry // 8 # lookup from daedlookup.xls
737 for row in range(rows):
738 current[row][col] = frame[row][2 * entry]
739 current[row][col + 256] = frame[row][2 * entry + 1]
740
741 for row in range(rows):
742 mapped[row][0:32] = current[row][320:352]
743 mapped[row][32:64] = current[row][352:384]
744 mapped[row][64:96] = current[row][192:224]
745 mapped[row][96:128] = current[row][160:192]
746 mapped[row][128:160] = current[row][256:288]
747 mapped[row][160:192] = current[row][288:320]
748 mapped[row][192:224] = current[row][416:448]
749 mapped[row][224:256] = current[row][32:64]
750 mapped[row][256:288] = current[row][128:160]
751 mapped[row][288:320] = current[row][224:256]
752 mapped[row][320:352] = current[row][384:416]
753 mapped[row][352:384] = current[row][448:480]
754 mapped[row][384:416] = current[row][480:512]
755 mapped[row][416:448] = current[row][0:32]
756 mapped[row][448:480] = current[row][64:96]
757 mapped[row][480:512] = current[row][96:128]
758 parsed.append(mapped)
759
760 images = self.caca.partition(parsed, columns)
761 flatimages = [flattenlist(x) for x in images]
762 return flatimages
