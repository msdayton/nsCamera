56* Must be called before other ZestETM1 functions. *

58ZESTETM1_STATUS ZestETM1Init(void)
59{
60#if defined(MSVC) || defined(WINGCC)
61 WORD VersionRequested;
62 WSADATA WSAData;
63 int Error;
64
65 VersionRequested = MAKEWORD(2, 2);
66 Error = WSAStartup(VersionRequested, &WSAData);
67 if (Error!=0)
68 {
69 ZESTETM1_ERROR_GENERAL("ZestETM1Init", ZESTETM1_SOCKET_ERROR);
70 }
71
72 // Confirm that the WinSock DLL supports 2.2.
73 // Note that if the DLL supports versions greater
74 // than 2.2 in addition to 2.2, it will still return
75 // 2.2 in Version since that is the version we
76 // requested.
77 if (LOBYTE(WSAData.wVersion)!=2 ||
78 HIBYTE(WSAData.wVersion)!=2)
79 {
80 WSACleanup( );
81 ZESTETM1_ERROR_GENERAL("ZestETM1Init", ZESTETM1_SOCKET_ERROR);
82 }
83#endif
84
85 return ZESTETM1_SUCCESS;
86}
