149 def setTiming(self, side="AB", sequence=None, delay=0):
150 """
151 Sets timing registers based on 'sequence.' Requesting (0,0) timing will clear the
152 timing register.
153 *WARNING* if the entire sequence does not fit into the 40-bit register space,
154 then the actual timings generated may differ from those requested. If the
155 timing sequence fits only once into the register space (i.e., for a single
156 frame, open + closed > 20 ns), then the actual timing will be (n, 40-n),
157 irrespective of the setting of second parameter, e.g. (35,1) will actually
158 result in (35,5) timing.
159 *NOTE* Icarus sensors generally cannot use 1 ns timing, so all values for these
160 devices (besides the delay) should be at least 2 ns
161
162 Args:
163 side: Hemisphere 'A', 'B', 'AB'
164 sequence: two-element tuple of timing durations in ns, e.g., '(5,2)'
165 delay: initial delay in ns (1 ns delay is acceptable)
166
167 Returns:
168 tuple (error string, 10-character hexadecimal representation of timing
169 sequence)
170 """

172 "{}setTiming; side={}, sequence={}, delay={}".format(
173 self.loginfo, side, sequence, delay
174 )

177 sequence = (3, 2)

187 + str(delay)

194 + side

202 return err1 + err2, full40hex
203 if side.upper() == "A":
204 lowreg = "HS_TIMING_DATA_ALO"
205 highreg = "HS_TIMING_DATA_AHI"
206 elif side.upper() == "B":
207 lowreg = "HS_TIMING_DATA_BLO"
208 highreg = "HS_TIMING_DATA_BHI"

211 self.logerr

273 actual = self.getTiming(side, actual=True)

434 def getTiming(self, side, actual):
435 """
436 actual = True: returns actual high speed intervals that will be generated by the
437 FPGA as list
438 False: Returns high speed timing settings as set by setTiming. Assumes
439 that timing was set via the setTiming method--it will not accurately
440 report arbitrary timings set by direct register sets or manual
441 shutter control
442
443 Args:
444 side: Hemisphere 'A' or 'B'
445 actual: False: return HST settings
446 True: calculate and return actual HST behavior
447
448 Returns:
449 actual= True: list of shutter intervals;
450 icarus: [delay, open0, closed0, open1, closed1, open2, closed2,
451 open3]
452 daedalus: [delay, open0, closed0, open1, closed1, open2]
453 False: tuple (hemisphere label,
454 'open shutter' in ns,
455 'closed shutter' in ns,
456 initial delay in ns)
457
458 """

461 side = "A"

463 if side.upper() == "A":
464 lowreg = "HS_TIMING_DATA_ALO"
465 highreg = "HS_TIMING_DATA_AHI"
466 elif side.upper() == "B":
467 lowreg = "HS_TIMING_DATA_BLO"
468 highreg = "HS_TIMING_DATA_BHI"
469 else:
470 logging.error(
471 self.logerr
472 + "Invalid sensor side: "
473 + side
474 + "; timing settings unchanged"
475 )
476 return "", 0, 0, 0

482 "returning zeroes "
483 )
484 return side.upper(), 0, 0, 0
485 full40hex = highpart[-2:] + lowpart.zfill(8)
486 full40bin = "{0:0=40b}".format(int(full40hex, 16))
487 logging.debug(self.logdebug + "full40bin = " + str(full40bin))
488 if actual:

492 else:

537 def setManualTiming(self, timing=None):
538 """
539 Manual shutter timing, seven intervals to assign to both hemispheres, e.g.,
540 [(100,150,100,150,100,150,100)] for frame 0 open for 100 ns, an interframe
541 pause of 50 ns,frame 1 open for 100 ns, etc.
542 Provide two sets of seven intervals, e.g., [(100,150,100,150,100,150,100),
543 (200,250,200,250,200,250,200)] to program the A and B hemispheres
544 independently
545
546 Overridden in daedalus.py
547
548 The timing list is flattened before processing; the suggested tuple structure is
549 just for clarity (first tuple is A, second is B) and is optional.
550
551 The actual timing is rounded down to the nearest multiple of 25 ns. (Each
552 count = 25 ns. e.g., a request for 140 ns rounds down to a count of '5',
553 which corresponds to 125 ns))
554 - Minimum timing is 75 ns
555 - Maximum is 25 * 2^30 ns (approximately 27 seconds)
556
557 Args:
558 timing: 7- or 14-element list (substructure optional) in nanoseconds
559
560 Returns:
561 tuple (error string, response string from final message)
562 """
563 if timing is None:
564 logging.info(
565 self.loginfo
566 + "No manual timing setting provided, defaulting to (100, 150, 100, "
567 " 150, 100, 150, 100) for both hemispheres"
568 )
569 timing = [
570 (100, 150, 100, 150, 100, 150, 100),
571 (100, 150, 100, 150, 100, 150, 100),
572 ]
573 logging.info(self.loginfo + "Manual shutter sequence: " + str(timing))
574 flattened = flattenlist(timing)
575 if len(flattened) == 7:
576 flattened = 2 * flattened
577 if (
578 len(flattened) != 14
579 or not all(isinstance(x, numbers.Real) for x in flattened)
580 or not all(x >= 75 for x in flattened)
581 or not all(x <= 26843545600 for x in flattened)
582 ):
583 err = self.logerr + "Invalid manual shutter timing list: " + str(timing)
584 logging.error(err + "; timing settings unchanged")
585 return err, "00000000"
586
587 timecounts = [int(a // 25) for a in flattened]
588 self.ca.sensmanual = timing
589 self.ca.senstiming = {} # clear HST settings from ca object
590
591 control_messages = [
592 ("W0_INTEGRATION", "{0:#0{1}x}".format(timecounts[0], 10)[2:10]),
593 ("W0_INTERFRAME", "{0:#0{1}x}".format(timecounts[1], 10)[2:10]),
594 ("W1_INTEGRATION", "{0:#0{1}x}".format(timecounts[2], 10)[2:10]),
595 ("W1_INTERFRAME", "{0:#0{1}x}".format(timecounts[3], 10)[2:10]),
596 ("W2_INTEGRATION", "{0:#0{1}x}".format(timecounts[4], 10)[2:10]),
597 ("W2_INTERFRAME", "{0:#0{1}x}".format(timecounts[5], 10)[2:10]),
598 ("W3_INTEGRATION", "{0:#0{1}x}".format(timecounts[6], 10)[2:10]),
599 ("W0_INTEGRATION_B", "{0:#0{1}x}".format(timecounts[7], 10)[2:10]),
600 ("W0_INTERFRAME_B", "{0:#0{1}x}".format(timecounts[8], 10)[2:10]),
601 ("W1_INTEGRATION_B", "{0:#0{1}x}".format(timecounts[9], 10)[2:10]),
602 ("W1_INTERFRAME_B", "{0:#0{1}x}".format(timecounts[10], 10)[2:10]),
