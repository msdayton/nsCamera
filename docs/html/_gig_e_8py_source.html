27class GigE:
28 """
29 Code to manage Gigabit Ethernet connection to board. Each GigE object manages a
30 single OT card; to use multiple cards, instantiate multiple cameraAssembler
31 objects, each specifying the unique IPs of the corresponding OT card.
32
33 Note: Orange Tree card must be configured before use. See the README for details
34
35 Exposed methods:
36 arm() - puts camera into wait state for external trigger
37 readFrames() - waits for data ready register flag, then copies camera image data
38 into numpy arrays
39 readoff() - waits for data ready register flag, then copies camera image data
40 into numpy arrays; returns payload, payload size, and error message
41 sendCMD(pkt) - sends packet object via serial port
42 readSerial(size, timeout) - read 'size' bytes from connection
43 writeSerial(outstring) - submits string 'outstring' over connection
44 closeDevice() - close connections and free resources
45 getCardIP() - returns IP address of OT card
46 getCardInfo() - prints report of details of OT card and connection
47 """

170 def sendCMD(self, pkt):
171 """
172 Submit packet and verify the response packet.
173 Packet communications with FPGA omit CRC suffix, so adds fake CRC bytes to
174 response
175
176 Args:
177 pkt: Packet object
178
179 Returns:
180 tuple (error, response string)
181 """
182
183 pktStr = pkt.pktStr()[0:16]
184 logging.debug(self.logdebug + "sendCMD packet: " + str(pktStr))
185 err = ""
186 self.ca.writeSerial(pktStr)
187 if (
188 hasattr(self.ca, "board")
189 and pktStr[4] == "0"
190 and pktStr[5:8] == self.ca.board.registers["SRAM_CTL"]
191 ):
192 bufsize = self.payloadsize + 16
193 resptext = self.readSerial(bufsize)
194
195 if len(resptext) < 32:
196 logging.debug(self.logdebug + "sendCMD resptext = " + str(resptext))
197 else:
198 logging.debug(
199 self.logdebug
200 + "sendCMD resptext (truncated) = "
201 + str(resptext)[0:32]
202 )
203
204 if len(resptext) < bufsize + 16:
205 err += (
206 self.logerr + "sendCMD- packet too small, payload may be incomplete"
207 )
208 logging.error(err)
209 else:
210 # workaround for initial setup before board object has been initialized
211 resp = self.readSerial(8)
212 logging.debug(self.logdebug + "sendCMD response: " + str(resp))
213 if len(resp) < 8:
214 err += self.logerr + "sendCMD- response too small, returning zeros"
215 resptext = "00000000000000000000"
216 logging.error(err)
217 else:
218 resptext = resp + "0000"
219 return err, resptext

225
226 Args:

232 tuple (error, response string)

248 def readFrames(self, waitOnSRAM, timeout=0, fast=False, columns=1):
249 """
250 Copies image data from board into numpy arrays.
251
252 Args:
253 waitOnSRAM: if True, wait until SRAM_READY flag is asserted to begin copying
254 data
255 timeout: passed to waitForSRAM; after this many seconds begin copying data
256 irrespective of SRAM_READY status; 'zero' means wait indefinitely
257 WARNING: If acquisition fails, the SRAM will not contain a current image,
258 but the code will copy the data anyway
259 fast: if False, parse and convert frames to numpy arrays; if True, return
260 unprocessed text stream
261 columns: 1 for single image per frame, 2 for separate hemisphere images
262
263 Returns:
264 list of numpy arrays OR raw text stream
265
266 """
267 frames, _, _ = self.readoff(waitOnSRAM, timeout, fast, columns)
268 return frames

270 def readoff(self, waitOnSRAM, timeout=0, fast=False, columns=1):
271 """
272 Copies image data from board into numpy arrays; returns data, length of data,
273 and error messages. Use 'readFrames()' unless you require this additional
274 information
275
276 Args:
277 waitOnSRAM: if True, wait until SRAM_READY flag is asserted to begin copying
278 data
279 timeout: passed to waitForSRAM; after this many seconds begin copying data
280 irrespective of SRAM_READY status; 'zero' means wait indefinitely
281 WARNING: If acquisition fails, the SRAM will not contain a current image,
282 but the code will copy the data anyway
283 fast: if False, parse and convert frames to numpy arrays; if True, return
284 unprocessed text stream
285 columns: 1 for single image per frame, 2 for separate hemisphere images
286
287 Returns:
288 tuple (list of numpy arrays OR raw text stream, length of downloaded payload
289 in bytes, payload error flag) since CRC check is handled by TCP/IP,
290 payload error flag is always False for GigE
291 """
292 logging.info(self.loginfo + "readoff")
293 logging.debug(
294 self.logdebug
295 + "readoff: waitonSRAM = "
296 + str(waitOnSRAM)
297 + "; timeout = "
298 + str(timeout)
299 + "; fast = "
300 + str(fast)
301 )
302 # Wait for data to be ready on board
303 # Skip wait only if explicitly tagged 'False' ('None' defaults to True)
304 if waitOnSRAM is not False:
305 self.ca.waitForSRAM(timeout)
306 self.skipError = False
307 err, rval = self.ca.readSRAM()
308 if err:
309 logging.error(self.logerr + "Error detected in readSRAM")
310 elif self.ca.boardname == "llnl_v4":
311 # self.ca.setSubregister('SWACK','1')
312 pass
313 # extract the data. Remove header; the FPGA returns a packet without the CRC
314 # suffix
315 logging.debug(self.logdebug + "readoff: first 64 chars: " + str(rval[0:64]))
316 data = rval[32:]
317 if fast:
318 return data, len(data) // 2, bool(err)
319 else:
320 parsed = generateFrames(self.ca, data, columns)
321 return parsed, len(data) // 2, bool(err)

324 """
325 Transmit string to board
326
327 Args:
328 outstring: string to write
329 timeout: serial timeout in sec (defaults to self.writeTimeout)
330
331 Returns:
332 integer number of bytes written

334 logging.debug(

342 timeout = self.writeTimeout

364 """

402 def openDevice(self, timeout=30):
403 """
404 Find Orange Tree card and open a connection; if IP is supplied as parameter for
405 the CameraAssembler, bypass network search and connect directly to indicated
406 IP address
407
408 Args:
409 timeout: timeout in seconds for attempting to connect to a card
410 """
411 err = self._zest.ZestETM1Init()
412 if err:
413 logging.critical(self.logcrit + "ZestETM1Init failure")
414 sys.exit(1)
415 logging.info(self.loginfo + "searching for Orange Tree cards")
416 NumCards = C.c_ulong(0)
417
418 if self.ca.iplist:
419 ubyte4 = C.c_ubyte * 4
420 self.CardInfo.IPAddr = ubyte4(*self.ca.iplist)
421 self.CardInfo.ControlPort = C.c_ushort(self.dport)
422 self.CardInfo.Timeout = C.c_ulong(self.writeTimeout)
423 self.closecard = False
424 else:
425 wait = 0
426 while True:
427 logging.debug(
428 self.logdebug + "openDevice: connection wait = " + str(wait)
429 )
430 err = self.ZCountCards(C.byref(NumCards), C.byref(self.CardInfoP), 1000)
431 if err:
432 logging.critical(self.logcrit + "CountCards failure")
433 sys.exit(1)
434 if NumCards.value > 0:
435 break
436 if wait == timeout:
437 logging.critical(
438 "{}No Orange Tree cards found in {} seconds".format(
439 self.logcrit, timeout
440 )
441 )
442 sys.exit(1)
443 elif not wait % 5:
444 logging.info(
445 "{}Still trying to connect after {} seconds...".format(
446 self.loginfo, wait
447 )
448 )
449 wait += 1
450 logging.info(
451 self.loginfo + str(NumCards.value) + " Orange Tree card(s) found"
452 ) # TODO: add check for GigE bit in board description
453 err = self.ZOpenConnection(
454 self.CardInfoP, 0, self.dport, 0, C.byref(self.Connection)
455 )
456 if err:
457 if err == 0x4000:
458 logging.warning(
459 self.logerr + "OT Card emitted an undefined warning message"
460 )
461 else:
462 logging.critical(
463 self.logcrit + "OpenConnection failure: " + self.ZErrorDict[err]
464 )
465 sys.exit(1)
