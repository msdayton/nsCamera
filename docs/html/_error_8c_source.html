81/******************************************************************************
82* Get a human-readable error string for a status code *
83******************************************************************************/
84ZESTETM1_STATUS ZestETM1GetErrorMessage(ZESTETM1_STATUS Status,
85 char **Buffer)
86{
87 if (Status>ZESTETM1_MAX_ERROR ||
88 (Status<ZESTETM1_ERROR_BASE && Status>=ZESTETM1_MAX_WARNING) ||
89 (Status<ZESTETM1_WARNING_BASE && Status>=ZESTETM1_MAX_INFO))
90 {
91 return ZESTETM1_ILLEGAL_STATUS_CODE;
92 }
93
94 *Buffer = ZESTETM1_ERROR_STRING(Status);
95 return ZESTETM1_SUCCESS;
96}
