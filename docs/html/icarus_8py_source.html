121 def checkSensorVoltStat(self):
122 """
123 Checks register tied to sensor select jumpers to confirm match with sensor
124 object
125
126 Returns:
127 boolean, True if jumpers select for Icarus sensor
128 """
129 logging.debug(self.logdebug + "checkSensorVoltStat")
130 err, status = self.caca.getSubregister("ICARUS_DET")
131 if err:
132 logging.error(self.logerr + "unable to confirm sensor status")
133 return False
134 if not int(status):
135 logging.error(self.logerr + "Icarus sensor not detected")
136 return False
137 return True
