506 def waitForSRAM(self, timeout):
507 """
508 Wait until subreg 'SRAM_READY' flag is true or timeout is exceeded;
509 timeout = None or zero means wait indefinitely
510
511 Args:
512 timeout - time in seconds before readoff proceeds automatically without
513 waiting for SRAM_READY flag
514
515 Returns:
516 error string
517 """
518 logging.info(self.loginfo + "waitForSRAM, timeout = " + str(timeout))
519 waiting = True
520 starttime = time.time()
521 err = ""
522 while waiting:
523 err, status = self.ca.getSubregister("SRAM_READY")
524 if err:
525 err = self.logerr + "error in register read: " + err + " (waitForSRAM)"
526 logging.error(err)
527 if int(status):
528 waiting = False
529 logging.info(self.loginfo + "SRAM ready")
530 if self.ca.abort:
531 waiting = False
532 logging.info(self.loginfo + "readoff aborted by user")
533 self.ca.abort = False
534 if timeout and time.time() - starttime > timeout:
535 err += self.logerr + "SRAM timeout; proceeding with download attempt"
536 logging.error(err)
537 return err
538 # Slow down for debugging (avoid thousands of messages)
539 if self.ca.verbose >= 5:
540 time.sleep(0.5)
541 return err

628 def getTemp(self, scale=None):
629 """
630 Read temperature sensor
631 Args:
632 scale: temperature scale to report (defaults to C, options are F and K)
633
634 Returns:
635 temperature as float on given scale
636 """
637 err, rval = self.ca.getMonV("MON_TEMP", errflag=True)
638 if err:
639 logging.error(
640 self.logerr + "unable to retrieve temperature information ("
641 'getTemp), returning "0" '
642 )
643 return 0.0

653 def getPressure(self, offset=None, sensitivity=None, units=None):
654 """
655 Read pressure sensor. Uses default offset and sensitivity defined in init
656 function unless alternatives are specified. NOTE: to reset defaults, reassign
657 board.defoff and board.defsens explicitly
658
659 Args:
660 offset: non-default offset in mv/V
661 sensitivity: non-default sensitivity in mV/V/span
662 units: units to report pressure (defaults to Torr, options are psi, bar,
663 inHg, atm)
664
665 Returns:
666 Pressure as float in chosen units, defaults to torr
667 """
668 if offset is None:
669 offset = self.defoff
670 if sensitivity is None:
671 sensitivity = self.defsens
672 if units is None:
673 units = "torr"
674 pplus = self.ca.getMonV("MON_PRES_PLUS")
675 pminus = self.ca.getMonV("MON_PRES_MINUS")
676 delta = 1000 * (pplus - pminus)
677 ratio = sensitivity / 30 # nominal is 21/30
678 psi = (delta - offset) / ratio
679 if units.lower() == "psi":
680 press = psi
681 elif units.lower() == "bar":
682 press = psi / 14.504
683 elif units.lower() == "atm":
684 press = psi / 14.695
685 elif units.lower() == "inHg":
686 press = psi * 2.036
687 else:
688 press = 51.715 * psi # default to Torr
689
690 return press
